约束

约束是使用在列上，用于约束列的

参考 https://blog.csdn.net/f45056231p/article/details/81070437

主键、外键和索引的区别？

主键 外键 索引
定义： 唯一标识一条记录，不能有重复的，不允许为空 表的外键是另一表的主键, 外键可以有重复的, 可以是空值 该字段没有重复值，但可以有一个空值
作用： 用来保证数据完整性 用来和其他表建立联系用的 是提高查询排序的速度
个数： 主键只能有一个 一个表可以有多个外键 一个表可以有多个惟一索引

1. 主键约束（唯一标识）

关系型数据库中的一条记录中有若干个属性，若其中某一个属性组(注意是组)能唯一标识一条记录，该属性组就可以成为一个主键

  • 特点：唯一、非空、可引用
-- 设置主键
-- 方法一
create table stu (
  sid int primary key,
  name varchar(50),
  age int
) ;
-- 方法二
create table stu (
	sid int,
  name varchar(50),
  age int,
  primary key (sid)
);
-- 方法三
alter table stu add primary key(sid);

-- 删除主键
alter table stu drop primary key;

2. 主键自增长

  • 使用auto_increment实现自增长，存记录时如果设置主键值为null，则自动自增，因此真实主键值不为空；
  • 主键需要是整型int；
create table stu(
  sid int primary key auto_increment,
  name varchar(50),
  age int
);

-- 此时插入记录时候可以使用null，但是auto_increment会自动实现自增
insert into stu values (
  null, 'zhangSan', 18
);

3. 非空约束

  • 使用NOT NULL实现非空约束
create table stu (
  sid int primary key,
  name varchar(50) not null,
  age int
);

4. 唯一性约束

  • 使用UNIQUE实现唯一性约束
  • 可以和非空约束同时存在
create table stu (
  sid int primary key,
  name varchar(50) not null unique,
  age int not null
);

5. 概念模型

  • 对象模型：类似于java中的domain，例如User、Student
  • 关系模型：数据库中的表的概念（只能多方引用一方，而且只能引用主键，而不是一整行记录）
    • 1对1
    • 1对多
    • 多对多
  • 互相引用

6. 外键

  • 外键必须是另一张表（或本表）的主键
  • 外键可以不唯一
  • 外键可以重复
  • 外键可以为空

7. 数据库的对应关系

7.1 多对一关系（最常见）

-- 下面的dept和emp是一个“多对一”的例子
create table if not exists dept(
		deptno int primary key,
    dname varchar(50),
    loc varchar(50)
);

-- 这里的外键deptno引用的是表dept的主键
-- 外键mgr引用的是自身表中的主键
create table if not exists emp(
		empno int primary key,
    ename varchar(50),
    job varchar(50),
    mgr int,
    hiredate date,
    sal decimal(7,2),
    COMM decimal(7,2),
    deptno int,
    constraint fk_emp foreign key(mgr) references emp(empno),
    constraint fk_emp_dept foreign key(deptno) references dept(deptno)
);

7.2 一对一关系

-- 如何实现一对一的关系模型？
-- 将外键设置为自身主键，然后引用另一个表的主键
create table if not exists hasband (
		hid int primary key auto_increment,
  	hname varchar(50)
);
insert into hasband values (null, 'zhangSan');
insert into hasband values (null, 'liSi');
-- 由于auto_increment的作用，这里设置的null会自动填充

create table if not exists wife (
		wid int primary key auto_increment,
  	wname varchar(50)
);
-- 可以通过alter修改表操作来设置表的外键
alter table wife 
add constraint fk_wife_hasband foreign key(wid) references hasband(hid);
-- 这里将主键wid设置为外键，并引用hasband的主键hid，就能实现唯一的hid/wid对应一对夫妻，即一对一关系
insert into wife values (1, 'wangWu');
insert into wife values (2, 'zhaoLiu');
-- 这里由于主键设置成了外键，因此其必须需要满足唯一性、非空性、引用hid（由于自增效果，这里hid从1开始）

7.3 多对多关系

-- 实现多对多的关系模型
-- 需要使用第三张表充当“中间表”，在中间表中使用两个外键，每个外键分别引用其他两个表的主键
create table stu (
		sid int primary key auto_increment,
  	sname varchar(50)
);
insert into stu values (null, 'zhangSan');
insert into stu values (null, 'liSi');
insert into stu values (null, 'wangWu');
insert into stu values (null, 'zhaoLiu');

create table teacher (
		tid int primary key auto_increment,
  	tname varchar(50)
);
insert into teacher values (null, 'Bob');
insert into teacher values (null, 'Jack');
insert into teacher values (null, 'Rose');

-- 中间表
create table stu_tea (
		sid int,
  	tid int,
  	constraint fk_stu foreign key(sid) references stu(sid),
  	constraint fk_tea foreign key(tid) references tea(tid)
);
insert into stu_tea values (1, 1);
insert into stu_tea values (3, 1);
insert into stu_tea values (4, 1);
insert into stu_tea values (1, 2);
insert into stu_tea values (2, 2);
insert into stu_tea values (2, 3);
insert into stu_tea values (4, 3);
